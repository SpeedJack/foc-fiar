\section{Peer-to-Peer messages}\label{sec:p2pmsgs}

\begin{itemize}
	\item[] \standout{DHKEY} \emph{(signed)} The message sent by one player
		to the other for sending its own \emph{Diffie-Hellman} public
		key.
	\item[] \standout{GAME\_MOVE} \emph{(encrypted)} The message sent by one
		player to the other containing its game move.
	\item[] \standout{ERROR} The message sent from a client to the other
		containing an error code and an error message.
\end{itemize}
